Intégration ultra-rapide

Exemple Node.js / TypeScript 
import { FidelityPay } from '@fidelity-pay/sdk';

const fidelity = new FidelityPay('{{ firstKey() }}');

// Initialiser un paiement
const session = await fidelity.payments.create({
  amount: 5000,
  currency: 'XOF',
  country: 'SN',        // Code pays (SN, CI, BJ, etc.)
  operator: 'WAVE',     // Opérateur dynamique (WAVE, ORANGE, MIXX, etc.)
  phone: '77XXXXXXX',   // Requis pour le paiement direct (Push)
  description: 'Achat iPhone 15 Pro',
  customer: {
    email: 'client@example.com',
    name: 'Jean Dupont'
  },
  callback_url: 'https://votre-site.com/webhook'
});

console.log('ID Transaction:', session.id_transaction);
